Reconnecting Frontend to Backend: Polls Feature

1. Update Poll Model

First, ensure your Poll model in Swift matches your Django model:

struct Poll: Identifiable, Codable {
    let id: Int
    let title: String
    let options: [String]
    let createdAt: String
    let votes: [String: Int]?  // Add this if you want to track votes
    
    enum CodingKeys: String, CodingKey {
        case id
        case title
        case options
        case createdAt = "created_at"
        case votes
    }
}

2. Update PollsViewModel

Update your PollsViewModel to handle authentication and proper API calls:

class PollsViewModel: ObservableObject {
    @Published var polls: [Poll] = []
    private let baseURL = "http://localhost:8000"
    
    func fetchPolls(for circleId: Int) {
        guard let url = URL(string: "\(baseURL)/api/polls/?circle_id=\(circleId)") else {
            print("‚ùå Invalid URL")
            return
        }
        
        var request = URLRequest(url: url)
        request.setValue("application/json", forHTTPHeaderField: "Accept")
        
        // Add authentication token
        if let token = UserDefaults.standard.string(forKey: "auth_token") {
            request.setValue("Token \(token)", forHTTPHeaderField: "Authorization")
        }
        
        URLSession.shared.dataTask(with: request) { [weak self] data, response, error in
            if let error = error {
                print("‚ùå Network error: \(error)")
                return
            }
            
            guard let httpResponse = response as? HTTPURLResponse else {
                print("‚ùå Invalid response type")
                return
            }
            
            print("üì° Response status: \(httpResponse.statusCode)")
            
            guard let data = data else {
                print("‚ùå No data received")
                return
            }
            
            // Print raw response for debugging
            if let responseStr = String(data: data, encoding: .utf8) {
                print("üì¶ Raw response: \(responseStr)")
            }
            
            do {
                let decodedPolls = try JSONDecoder().decode([Poll].self, from: data)
                DispatchQueue.main.async {
                    self?.polls = decodedPolls
                    print("‚úÖ Loaded \(decodedPolls.count) polls")
                }
            } catch {
                print("‚ùå Decoding error: \(error)")
            }
        }.resume()
    }
    
    func createPoll(title: String, options: [String], circleId: Int) {
        guard let url = URL(string: "\(baseURL)/api/polls/") else { return }
        
        var request = URLRequest(url: url)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        
        if let token = UserDefaults.standard.string(forKey: "auth_token") {
            request.setValue("Token \(token)", forHTTPHeaderField: "Authorization")
        }
        
        let pollData = [
            "title": title,
            "options": options,
            "circle_id": circleId
        ] as [String : Any]
        
        do {
            request.httpBody = try JSONSerialization.data(withJSONObject: pollData)
        } catch {
            print("‚ùå Error encoding poll data: \(error)")
            return
        }
        
        URLSession.shared.dataTask(with: request) { [weak self] data, response, error in
            if let error = error {
                print("‚ùå Error creating poll: \(error)")
                return
            }
            
            guard let httpResponse = response as? HTTPURLResponse else { return }
            print("üì° Create poll response: \(httpResponse.statusCode)")
            
            if httpResponse.statusCode == 201 {
                self?.fetchPolls(for: circleId)
            }
        }.resume()
    }
}

Important Notes:

  • The authentication token should be stored in UserDefaults when the user logs in
  • The token format should match what your Django backend expects (usually "Token" prefix)
  • Make sure CORS is properly configured on your Django backend
  • The circle_id parameter must match what your backend expects

Testing the Connection

  1. First, verify your token is stored:
    print("üîë Auth Token:", UserDefaults.standard.string(forKey: "auth_token") ?? "No token found")
  2. Test the API directly:
    curl -H "Authorization: Token YOUR_TOKEN" http://localhost:8000/api/polls/?circle_id=CIRCLE_ID
  3. Check Django server logs for any errors
  4. Monitor Xcode console for the debug prints we added

Common Issues:

  • 401 Unauthorized: Check if token is being sent and is valid
  • 404 Not Found: Verify API endpoint URL is correct
  • Empty response: Make sure circle_id is correct and polls exist
  • Decoding errors: Check if response JSON matches Poll struct
